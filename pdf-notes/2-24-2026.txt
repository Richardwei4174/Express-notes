app.put("/api/users/:id"), (request, response) => {
  const {body, params: {id}} = request;

  const parsedId = parseInt(id);
  if (isNaN(parsedId)) return response.sendStatus(400); // checks if parsedId is actually a num

  const findUserIndex = mockUsers.findIndex( (user) => user.id === parsedId);

  if (findUserIndex === -1) return response.sendStatus(404);

  mockUsers[findUserIndex] = { id: parsedId, ...body}; // kept the id the same

  return response.sendStatus(200);
}

Query Params
‘?’ at the end of a website denotes a query string. It’s for key value pairs. We can use ‘&’ to add more of those pairs.

Essentially it means we get some kind of key value pair to the server side and we can do something about it. For instance we can do 
http://localhost:3000/?filter=username&value=an
Which would create a json request {filter: ‘username’, value:’an’} where we can decide what to do with those values.

Let’s say using the example above, we can check if filter and value “exist” as in they are on the website route (in this case yes) and than filter our our premade list of username.

Say we have a list of username and displayName as such:

const mockUsers = [
  {id: 1, username: "anson", displayName: "Anson"},
  {id: 2, username: "bobl", displayName: "Bob2"},
  {id: 3, username: "demonslayer3000", displayName: "Bob3"},
];

// we can filter out our list based on the value of “value” which is “an” in this example. It would go through the list and return back the list with “an” in the username

app.get('/', (request, response)=>{
  console.log(request.query);
  const{ query: {filter, value} } = request;
  
  if (filter && value) return response.send(
    mockUsers.filter( (user) => // no {} else it won't work
      // filter al the filter object that matches value
      user[filter].includes(value)
    )
  )
  // return normal is there's no filter AND value in the route
  return response.send(mockUsers);

});
